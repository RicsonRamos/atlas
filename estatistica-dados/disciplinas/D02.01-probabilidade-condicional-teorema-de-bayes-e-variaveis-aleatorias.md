## Implementação em Python
```python
import numpy as np
from scipy import stats

# 1. Resolução do clássico problema de Bayes em código
# Diagnóstico de Doença: Incidência = 1%, Sensibilidade = 99%, Falso Positivo = 5%
# Pergunta: Se o teste dá positivo, qual a probabilidade de ter a doença?
incidencia = 0.01          # P(D)
sensibilidade = 0.99       # P(Positivo | D)
falso_positivo = 0.05      # P(Positivo | ~D)

# P(~D) = 1 - P(D)
não_doente = 1 - incidencia 

# Teorema de Bayes
# P(D|Positivo) = (P(Positivo|D) * P(D)) / (P(Positivo|D)*P(D) + P(Positivo|~D)*P(~D))
prob_positivo = (sensibilidade * incidencia) + (falso_positivo * não_doente) # Probabilidade Total
prob_doente_dado_positivo = (sensibilidade * incidencia) / prob_positivo

print(f"Probabilidade total de teste positivo: {prob_positivo:.4f}")
print(f"P(Ter a Doença | Teste Positivo): {prob_doente_dado_positivo * 100:.2f}%")


# 2. Utilizando scipy.stats para explorar funções de uma variável aleatória
# Exemplo: Variável aleatória contínua (Normal com média 100 e desvio padrão 15)
va_normal = stats.norm(loc=100, scale=15)

# Calcular probabilidade de valor menor ou igual a 115: P(X <= 115) - CDF
prob_menor_115 = va_normal.cdf(115)
# Calcular a densidade de probabilidade exatamente no ponto 100 (Pico) - PDF
pdf_100 = va_normal.pdf(100)
# Calcular o percentil 90: Valor x tal que P(X <= x) = 0.90 - PPF (Inversa da CDF)
percentil_90 = va_normal.ppf(0.90)

print(f"P(X <= 115): {prob_menor_115:.4f}")
print(f"PDF no ponto 100: {pdf_100:.4f}")
print(f"Valor limite para o percentil 90 (PPF): {percentil_90:.2f}")
```
